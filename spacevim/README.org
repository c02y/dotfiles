# Before using this configs, remove old ~/.vim
$ cd ~/Dotfiles.d
$ stowsh -r vim
# Install space-vim
$ curl -fsSL https://raw.githubusercontent.com/liuchengxu/space-vim/master/install.sh | bash
# or shorter one
$ curl -fsSL https://git.io/vFUhE | bash
# For Vim
$ ln -s ~/.space-vim/init.vim ~/.vimrc
# For NeoVim
$ mkdir ~/.config/nvim; ln -s ~/.space-vim/init.vim ~/.config/nvim/init.vim; pip install neovim clang
# then create links to home directory
$ stowsh spacevim

# By default, sapce-vim installation instruction only use git clone --depth=1
# You may want to download all the logs after install space-vim using:
$ git pull --unshallow

# update space-vim
$ cd ~/.space-vim; make update
# uninstall space-vim
$ cd ~/.space-vim; make uninstall
